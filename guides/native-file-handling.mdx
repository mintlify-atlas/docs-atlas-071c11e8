---
title: "Native File Handling"
description: "Open, save, and drag files using native OS dialogs"
---

Electron's [`dialog`](/api/dialog) module lets you show native OS dialogs for opening files, saving files, and displaying message boxes. You can also enable dragging files out of your app's window directly into the operating system.

## Opening files

Use `dialog.showOpenDialog` to show a native file picker. It returns an object containing the selected file paths.

```javascript title="main.js"
const { dialog } = require('electron')

async function openFile () {
  const { canceled, filePaths } = await dialog.showOpenDialog({
    title: 'Open file',
    properties: ['openFile']
  })

  if (!canceled) {
    console.log('Selected file:', filePaths[0])
  }
}
```

### Selecting multiple files

```javascript title="main.js"
const { canceled, filePaths } = await dialog.showOpenDialog({
  properties: ['openFile', 'multiSelections']
})

if (!canceled) {
  console.log('Selected files:', filePaths)
}
```

### Selecting a directory

```javascript title="main.js"
const { canceled, filePaths } = await dialog.showOpenDialog({
  properties: ['openDirectory']
})
```

### Filtering by file type

Pass a `filters` array to restrict which files are shown. Each entry has a `name` and an array of `extensions` (without dots).

```javascript title="main.js"
const { canceled, filePaths } = await dialog.showOpenDialog({
  title: 'Open image',
  filters: [
    { name: 'Images', extensions: ['jpg', 'jpeg', 'png', 'gif', 'webp'] },
    { name: 'All files', extensions: ['*'] }
  ],
  properties: ['openFile']
})
```

### Open dialog properties reference

| Property | Description |
|----------|-------------|
| `openFile` | Allow file selection. |
| `openDirectory` | Allow directory selection. |
| `multiSelections` | Allow multiple paths to be selected. |
| `createDirectory` | Allow creating new directories in the dialog (macOS only). |
| `promptToCreate` | Prompt to create if the path doesn't exist (Windows only). |
| `showHiddenFiles` | Show hidden files in the dialog (macOS and Windows). |
| `noResolveAliases` | Return alias path instead of target path (macOS only). |

<Note>
  On Windows and Linux, you cannot combine `openFile` and `openDirectory` in the same dialog. If both are set, a directory selector is shown.
</Note>

## Saving files

Use `dialog.showSaveDialog` to show a native save dialog. It returns the file path the user chose, or `undefined` if the dialog was cancelled.

```javascript title="main.js"
const { dialog } = require('electron')
const fs = require('node:fs/promises')

async function saveFile (content) {
  const { canceled, filePath } = await dialog.showSaveDialog({
    title: 'Save file',
    defaultPath: 'document.txt',
    filters: [
      { name: 'Text files', extensions: ['txt'] },
      { name: 'All files', extensions: ['*'] }
    ]
  })

  if (!canceled && filePath) {
    await fs.writeFile(filePath, content, 'utf8')
    console.log('File saved to:', filePath)
  }
}
```

## Message boxes

Use `dialog.showMessageBox` to display a native alert or confirmation dialog.

```javascript title="main.js"
const { dialog, BrowserWindow } = require('electron')

async function confirmDelete (win) {
  const { response } = await dialog.showMessageBox(win, {
    type: 'warning',
    title: 'Confirm delete',
    message: 'Are you sure you want to delete this file?',
    detail: 'This action cannot be undone.',
    buttons: ['Cancel', 'Delete'],
    defaultId: 0,
    cancelId: 0
  })

  if (response === 1) {
    console.log('User confirmed delete')
  }
}
```

The `type` option controls the icon shown: `'none'`, `'info'`, `'error'`, `'question'`, or `'warning'`.

## Calling dialogs from the renderer

Dialogs must be opened from the main process. Use IPC to trigger them from renderer code.

```javascript title="preload.js"
const { contextBridge, ipcRenderer } = require('electron')

contextBridge.exposeInMainWorld('dialog', {
  openFile: () => ipcRenderer.invoke('dialog:open-file'),
  saveFile: (content) => ipcRenderer.invoke('dialog:save-file', content)
})
```

```javascript title="main.js"
const { ipcMain, dialog } = require('electron')

ipcMain.handle('dialog:open-file', async () => {
  const { canceled, filePaths } = await dialog.showOpenDialog({
    properties: ['openFile']
  })
  return canceled ? null : filePaths[0]
})

ipcMain.handle('dialog:save-file', async (event, content) => {
  const { canceled, filePath } = await dialog.showSaveDialog({
    defaultPath: 'output.txt'
  })
  if (!canceled && filePath) {
    const fs = require('node:fs/promises')
    await fs.writeFile(filePath, content, 'utf8')
    return filePath
  }
  return null
})
```

```javascript title="renderer.js"
const filePath = await window.dialog.openFile()
if (filePath) {
  console.log('User selected:', filePath)
}
```

## Native file drag and drop out of the app

You can let users drag files out of your app's window and into the operating system. This uses `webContents.startDrag` in response to the `dragstart` DOM event.

<Steps>
  <Step title="Expose the drag starter via contextBridge">
    In your preload script, expose an IPC sender that the renderer can call when a drag begins.

    ```javascript title="preload.js"
    const { contextBridge, ipcRenderer } = require('electron')

    contextBridge.exposeInMainWorld('electron', {
      startDrag: (fileName) => ipcRenderer.send('ondragstart', fileName)
    })
    ```
  </Step>
  <Step title="Add a draggable element in HTML">
    Mark an element as `draggable` and reference your renderer script.

    ```html title="index.html"
    <div draggable="true" id="drag-item">Drag me to your desktop</div>
    <script src="renderer.js"></script>
    ```
  </Step>
  <Step title="Trigger the drag from the renderer">
    Intercept `ondragstart` and call the exposed method, passing the file name you want to export.

    ```javascript title="renderer.js"
    document.getElementById('drag-item').ondragstart = (event) => {
      event.preventDefault()
      window.electron.startDrag('my-file.txt')
    }
    ```
  </Step>
  <Step title="Handle the drag in the main process">
    Listen for the IPC message and call `event.sender.startDrag` with the full file path and a drag icon.

    ```javascript title="main.js"
    const { ipcMain } = require('electron')
    const path = require('node:path')

    ipcMain.on('ondragstart', (event, fileName) => {
      event.sender.startDrag({
        file: path.join(__dirname, fileName),
        icon: path.join(__dirname, 'drag-icon.png')
      })
    })
    ```
  </Step>
</Steps>

## Dragging files into the app

For dragging files _into_ your app, use the standard [HTML Drag and Drop API](https://developer.mozilla.org/en-US/docs/Web/API/HTML_Drag_and_Drop_API) directly in the renderer. No Electron-specific code is needed.

```javascript title="renderer.js"
document.addEventListener('dragover', (event) => {
  event.preventDefault()
})

document.addEventListener('drop', (event) => {
  event.preventDefault()
  const files = Array.from(event.dataTransfer.files)
  files.forEach(file => {
    console.log('Dropped file path:', file.path)
  })
})
```
